AZ Tabs

Default settings

new AZTabs(
{
    azTabsId: "az-tabs-1",
    azTabsHeightStyle: "content",
    azTabsOpenEvent: "click"
    azTabsIndex: 0
});

azTabsId Type: String Default: ""
azTabsHeightStyle Type: String Default: "content" Values: "content" or "auto"
azTabsOpenEvent Type: String Default: "click" Values: "click" or "mouseover"
azTabsIndex Type: Integer Default: 0 Unit: Index
functionlib/azTabsActivate Type: Function
$.subscribe("functionlib/azTabsActivate", function (e, data)
{
    data.azTabsId
    data.azTabsActivated
    data.azTabsDeactivated
});

functionlib/azTabs Type: Function
$.subscribe("functionlib/azTabs", function (e, data)
{
    data.azTabsId
    data.azTabsJQElement
});
